# 2024-3-1-895-最长上升子序列.cpp
```
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=5005;
int a[N],f[N];
int n;
int main()
{
    cin>>n;
for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    int res=0;
for(int i=1;i<=n;i++)
{   f[i]=1;
    for(int j=1;j<i;j++)
    {
        if(a[i]>a[j])
            f[i]=max(f[i],f[j]+1);
        res=max(res,f[i]);
    }
}
cout<<res;
    return 0;
}
```
