# 2024-3-2-1210-连号区间数.cpp
```
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=10010;
int a[N];
int n,res;
int main()
{
cin>>n;
for (int i = 0; i < n; ++i)
{
    scanf("%d",&a[i]);
}
for (int i = 0; i < n; ++i)
{
    int manx=-N,minx=N;
    for (int j = i; j <n ; ++j)
    {
        manx=max(manx,a[j]);
        minx=min(minx,a[j]);
        if(manx-minx==j-i) res++;
}
    }
cout<<res;


    return 0;
}
```
